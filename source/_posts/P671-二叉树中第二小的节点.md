class Solution {
    int temp;
    int rootVal;
    public int findSecondMinimumValue(TreeNode root) {
      temp = -1;
        rootVal = root.val;
        dfs(root);
        return temp;
    }

    public void dfs(TreeNode node){
        if(node==null)return;
        if(temp!=-1&&node.val>temp){
            return ;
        }
        if(node.val>rootVal){
            temp = node.val;
        }
        dfs(node.left);
        dfs(node.right);
    }
}
